       || result.find((entry) => entry.priority === 1)

    expect(homepage).toBeDefined()
    expect(homepage!.priority).toBe(1)
  })

  it("includes lastModified on all entries", () => {
    const result = sitemap()

    for (const entry of result) {

    }
  })
})
